# Views API

Views let users save and recall custom filter, sort, column, and density configurations for any list surface in the UI. Each view is scoped to a surface (e.g. `tasks`, `documents`, `tables`) and one view per surface can be marked as the default.

## Quick Start

Create a saved view with filters, apply it as the default, then update it later:

<CodeExample ts={`// 1. Create a saved view for the tasks surface
const view: { id: string; surface: string; name: string; isDefault: boolean } =
  await fetch('/api/views', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      surface: 'tasks',
      name: 'High Priority Active',
      filters: { priority: [0, 1], status: ['running', 'queued'] },
      sorting: { column: 'priority', direction: 'asc' },
      columns: { visible: ['title', 'status', 'priority', 'assignedAgent'], order: ['priority', 'title', 'status', 'assignedAgent'] },
      density: 'compact',
      isDefault: true,
    }),
  }).then((r) => r.json());
// → { id: "view-4a2b-8c9d", surface: "tasks", name: "High Priority Active", isDefault: true }

// 2. List all views for the tasks surface
const views: Array<{ name: string; isDefault: boolean }> =
  await fetch('/api/views?surface=tasks').then((r) => r.json());
// Default view is always first in the list
console.log(\`\${views.length} saved views, default: \${views[0].name}\`);

// 3. Update the view — add a new filter without changing the rest
await fetch(\`/api/views/\${view.id}\`, {
  method: 'PATCH',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({
    filters: { priority: [0, 1], status: ['running', 'queued'], assignedAgent: 'claude-code' },
  }),
});

// 4. Delete a view when no longer needed
await fetch(\`/api/views/\${view.id}\`, { method: 'DELETE' });`} python={`import requests

# 1. Create a saved view for the tasks surface
response: requests.Response = requests.post(
    "http://localhost:3000/api/views",
    json={
        "surface": "tasks",
        "name": "High Priority Active",
        "filters": {"priority": [0, 1], "status": ["running", "queued"]},
        "sorting": {"column": "priority", "direction": "asc"},
        "columns": {
            "visible": ["title", "status", "priority", "assignedAgent"],
            "order": ["priority", "title", "status", "assignedAgent"],
        },
        "density": "compact",
        "isDefault": True,
    },
)
view: dict = response.json()
# -> { "id": "view-4a2b-8c9d", "surface": "tasks", "name": "High Priority Active", "isDefault": true }

# 2. List all views for the tasks surface
views: list[dict] = requests.get(
    "http://localhost:3000/api/views",
    params={"surface": "tasks"},
).json()
# Default view is always first in the list
print(f"{len(views)} saved views, default: {views[0]['name']}")

# 3. Update the view — add a new filter without changing the rest
requests.patch(
    f"http://localhost:3000/api/views/{view['id']}",
    json={
        "filters": {"priority": [0, 1], "status": ["running", "queued"], "assignedAgent": "claude-code"},
    },
)

# 4. Delete a view when no longer needed
requests.delete(f"http://localhost:3000/api/views/{view['id']}")`} />

### List Views

<EndpointCard method="GET" path="/api/views" description="List all saved views for a given surface. Results are ordered with the default view first, then alphabetically by name.">

  { name: 'name', type: 'string', required: true, description: 'View display name' },

Fetch all saved views for a surface — the default view is always listed first:

<CodeExample ts={`// List views and find the default
const views: Array<{ name: string; isDefault: boolean; density: string }> =
  await fetch('/api/views?surface=tasks').then((r) => r.json());

const defaultView = views.find((v) => v.isDefault);
if (defaultView) {
  console.log(\`Default view: \${defaultView.name} (\${defaultView.density})\`);
}
views.forEach((v) => console.log(\`  \${v.isDefault ? '*' : ' '} \${v.name}\`));`} python={`import requests

# List views and find the default
response: requests.Response = requests.get(
    "http://localhost:3000/api/views",
    params={"surface": "tasks"},
)
views: list[dict] = response.json()

default_view: dict | None = next((v for v in views if v["isDefault"]), None)
if default_view:
    print(f"Default view: {default_view['name']} ({default_view['density']})")
for v in views:
    marker: str = "*" if v["isDefault"] else " "
    print(f"  {marker} {v['name']}")`} />

### Create View

<EndpointCard method="POST" path="/api/views" description="Create a new saved view. If isDefault is true, any existing default for the same surface is automatically unset.">

<SchemaTable title="Request Body" fields={[
  { name: 'surface', type: 'string', required: true, description: 'UI surface (e.g. tasks, documents, tables)' },
  { name: 'name', type: 'string', required: true, description: 'View display name' },
  { name: 'filters', type: 'object', required: false, description: 'Filter configuration (stored as JSON)' },
  { name: 'sorting', type: 'object', required: false, description: 'Sort configuration (stored as JSON)' },
  { name: 'columns', type: 'object', required: false, description: 'Column visibility and ordering (stored as JSON)' },
  { name: 'density', type: 'enum', required: false, description: 'comfortable (default), compact, or spacious' },
  { name: 'isDefault', type: 'boolean', required: false, description: 'Set as default view for this surface' },
]} />

**Response** `201 Created` — New view object

## View Density Options

| Density | Description |
|---------|-------------|
| **comfortable** | Default spacing. Good for scanning. |
| **compact** | Reduced padding. Shows more rows per screen. |
| **spacious** | Extra breathing room. Good for detailed review. |
